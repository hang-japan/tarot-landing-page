        document.addEventListener('DOMContentLoaded', () => {

            const packageName = document.getElementById('package-name');
            const packageDuration = document.getElementById('package-duration');
            const packagePrice = document.getElementById('package-price');
            const selectedDurationInput = document.getElementById('selected-duration');
            const selectedPriceInput = document.getElementById('selected-price');
            const selectedPackageNameInput = document.getElementById('selected-package-name');
            const confirmationMessage = document.getElementById('confirmation-message');

            packageCards.forEach(card => {
                card.addEventListener('click', () => {
                    packageCards.forEach(c => c.classList.remove('border-[#8c6a96]'));
                    card.classList.add('border-[#8c6a96]');

                    const name = card.getAttribute('data-package-name');
                    const price = card.getAttribute('data-price');
                    const duration = card.getAttribute('data-duration');

                    packageName.textContent = name;
                    packageDuration.textContent = duration;
                    packagePrice.textContent = price;
                    selectedPackageNameInput.value = name;
                    selectedDurationInput.value = duration;
                    selectedPriceInput.value = price;
                    selectedPackageSection.classList.remove('hidden');
                });
            });

            form.addEventListener('submit', (e) => {
                e.preventDefault();

                const formData = new FormData(form);
                const data = Object.fromEntries(formData.entries());

                console.log("Dữ liệu đăng ký:", data);
                
                form.reset();
                selectedPackageSection.classList.add('hidden');
                confirmationMessage.classList.remove('hidden');
            });
        });
